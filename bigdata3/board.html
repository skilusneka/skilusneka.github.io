					// 可收缩元素
					$(' .shrink-close').click(function(){
						var _this=$(this);
						windowShrink(_this);
					});
					$(' .shrink-open').click(function(){
						console.log(1)
						var _this=$(this);
						windowShrinkReverse(_this);
					});
					var windowShrink=function(el){
						$(el).parents('.shrink-left').addClass('shrink-done');
						$(el).parents('.shrink-right').addClass('shrink-done');
					}
					var windowShrinkReverse=function(el){
						$(el).parents('.shrink-done').removeClass('shrink-done');
					}
